/**
 * Handle POST requests from Electron/BPP App
 */
function doPost(e) {
    try {
        if (!e || !e.postData || !e.postData.contents) {
            return jsonResponse({ success: false, message: 'Invalid Request: No payload received.' });
        }

        const params = JSON.parse(e.postData.contents);
        const action = params.action;
        const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

        // 1. Auto-update EXPIRED status based on current date
        updateExpiryStatuses(ss);

        // 2. Route Actions
        if (action === 'REGISTER_TRIAL') {
            return handleTrialRegistration(ss, params);
        } else if (action === 'ACTIVATE_LICENSE') {
            return handleLicenseActivation(ss, params);
        } else if (action === 'VALIDATE_STARTUP') {
            return handleStartupValidation(ss, params);
        } else if (action === 'GET_MESSAGES') {
            return fetchLatestDeveloperMessages(ss);
        } else if (action === 'TRACK_LOGIN') {
            return handleTrackLogin(ss, params);
        } else if (action === 'GET_PRICING') {
            var pricingData = getLicenceFeeData(ss);
            return ContentService.createTextOutput(
                JSON.stringify({ success: true, pricing: pricingData })
            ).setMimeType(ContentService.MimeType.JSON);
        } else {
            return ContentService.createTextOutput(
                JSON.stringify({ success: false, message: 'Unknown action' })
            ).setMimeType(ContentService.MimeType.JSON);
        }
    } catch (err) {
        return jsonResponse({ success: false, message: 'Server Processing Error: ' + err.toString() });
    }
}

/**
 * [NEW] [ADMIN] Generates Key, Updates Sheet, and Sends Email
 */
function processLicenseIssuance(d) {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('BPP_APP_License'); // Using correct sheet name
    if (!sheet) {
        SpreadsheetApp.getUi().alert("Error: Sheet 'BPP_APP_License' not found!");
        return;
    }

    // 1. Generate Unique Secure Key (BPP-XXXX-XXXX-XXXX)
    var key = "BPP-" + Math.random().toString(36).substring(2, 6).toUpperCase() + "-" +
        Math.random().toString(36).substring(2, 6).toUpperCase() + "-" +
        Math.random().toString(36).substring(2, 6).toUpperCase();

    // 2. Append Row to Sheet (Col A to J)
    // Headers: Key | Name | UserID | Mobile | Email | Start | Expiry | MachineID | Status | DataSize
    sheet.appendRow([key, d.name, d.uid, d.mobile, d.email, "", d.expiry, "", "PENDING", d.limit]);

    // 3. Send Professional Email Confirmation
    sendEmailNotification(d.email, "Your BharatPay Pro License is Ready!", "LICENSE_ISSUANCE", {
        userName: d.name,
        userID: d.uid,
        licenseKey: key,
        expiryDate: d.expiry
    });

    SpreadsheetApp.getUi().alert("✅ Success!\n\nUser: " + d.name + "\nLicense Key: " + key + "\n\nConfirmation email has been dispatched.");
}

/**
 * Startup Validation logic (Strict 3-Point Hardware Match)
 */
function handleStartupValidation(ss, params) {
    const config = getAppConfig(ss);
    const devCreds = getDeveloperCredentials(ss);

    // 1. Check BPP_APP_License (Paid)
    let sheet = ss.getSheetByName('BPP_APP_License');
    if (sheet) {
        let data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            const row = data[i];
            const rowUserID = String(row[2]).trim().toLowerCase();
            const rowEmail = String(row[4]).trim().toLowerCase();
            const rowMachineId = String(row[7]).trim().toLowerCase();

            if (rowUserID === params.userID?.trim().toLowerCase() &&
                rowEmail === params.email?.trim().toLowerCase() &&
                rowMachineId === params.machineId?.trim().toLowerCase()) {

                if (isExpired(String(row[6])) || String(row[8]).trim().toUpperCase() === "EXPIRED") {
                    return jsonResponse({ success: false, message: "License Expired. Contact ilcbala.BharatPayRoll@gmail.com" });
                }

                return jsonResponse({
                    success: true,
                    message: "License Validated Successfully",
                    data: {
                        dataSize: row[9] || 5000,
                        latestVersion: config.latestVersion,
                        downloadUrl: config.downloadUrl,
                        downloadUrlWin7: config.downloadUrlWin7,
                        adminUser: row[1] || row[2],
                        adminPass: row[10] || "admin123",
                        devUser: devCreds.devUser,
                        devPass: devCreds.devPass,
                        licenceFee: getLicenceFeeData(ss)
                    }
                });
            }
        }
    }

    // 2. Check Approved_Trial_Data (Trial)
    sheet = ss.getSheetByName('Approved_Trial_Data');
    if (sheet) {
        let data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            const row = data[i];
            const rowMachineId = String(row[5]).trim().toLowerCase();
            const rowEmail = String(row[3]).trim().toLowerCase();
            const rowUserID = String(row[1]).trim().toLowerCase();

            if (rowUserID === params.userID?.trim().toLowerCase() &&
                rowEmail === params.email?.trim().toLowerCase() &&
                rowMachineId === params.machineId?.trim().toLowerCase()) {

                if (isExpired(String(row[7])) || String(row[9]).trim().toUpperCase() === "EXPIRED") {
                    return jsonResponse({ success: false, message: "Trial Expired. Contact ilcbala.BharatPayRoll@gmail.com" });
                }

                return jsonResponse({
                    success: true,
                    message: "Trial Validated Successfully",
                    data: {
                        dataSize: row[8] || 50,
                        latestVersion: config.latestVersion,
                        downloadUrl: config.downloadUrl,
                        downloadUrlWin7: config.downloadUrlWin7,
                        adminUser: row[1] || "admin",
                        adminPass: row[2] || "admin123",
                        devUser: devCreds.devUser,
                        devPass: devCreds.devPass
                    }
                });
            }
        }
    }

    trackFailedAttempt(ss, params.machineId, params.email, "STARTUP_VIOLATION");
    return jsonResponse({ success: false, message: "Login Violation: Unauthorised Machine ID or Credentials." });
}

/**
 * Handles Trial Registration (Strict restoration logic for V02.0.05)
 */
function handleTrialRegistration(ss, params) {
    const sheet = ss.getSheetByName('Approved_Trial_Data');
    if (!sheet) return jsonResponse({ success: false, message: 'Critical Error: Approved_Trial_Data sheet missing.' });

    let data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
        const row = data[i];
        const rowEmail = String(row[3]).trim().toLowerCase();
        const rowMobile = String(row[4]).trim();
        const rowMachineId = String(row[5]).trim().toLowerCase();
        const currentStatus = String(row[9]).trim().toUpperCase(); // Col J

        if (rowEmail === params.email.trim().toLowerCase() &&
            rowMobile === String(params.mobile).trim()) {

            // 1. VIOLATION CHECK: Prevent using same account on a DIFFERENT machine
            if (rowMachineId && rowMachineId !== params.machineId.trim().toLowerCase()) {
                return jsonResponse({ success: false, message: 'Violation: This account is already registered on another machine.' });
            }

            // 2. RESTORATION CHECK: If already registered, return existing dates
            if (currentStatus === 'REGISTERED' && row[6] && row[7]) {
                return jsonResponse({
                    success: true,
                    message: '🛡️ Trial History Found: Your original 90-day trial has been restored for this machine.',
                    data: {
                        status: 'REGISTERED',
                        startDate: String(row[6]),
                        expiryDate: String(row[7]),
                        dataSize: row[8] || 50
                    }
                });
            }

            // 3. FRESH REGISTRATION
            const now = new Date();
            const startStr = Utilities.formatDate(now, "GMT+5:30", "dd-MM-yyyy");
            const trialDays = 90;
            const end = new Date(now.getTime() + (trialDays * 24 * 60 * 60 * 1000));
            const endStr = Utilities.formatDate(end, "GMT+5:30", "dd-MM-yyyy");

            sheet.getRange(i + 1, 1).setValue(params.userName);
            sheet.getRange(i + 1, 2).setValue(params.userID);
            sheet.getRange(i + 1, 6).setValue(params.machineId);
            sheet.getRange(i + 1, 7).setValue(startStr);
            sheet.getRange(i + 1, 8).setValue(endStr);
            sheet.getRange(i + 1, 10).setValue('REGISTERED');

            sendEmailNotification(params.email, "Welcome to BharatPay Pro - Trial Registration Successful", "TRIAL_SUCCESS", {
                expiryDate: endStr
            });

            // Notify Developer
            sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "New Trial Registration: " + params.userName, "DEV_TRIAL_SUCCESS", {
                userName: params.userName,
                userID: params.userID,
                email: params.email,
                machineId: params.machineId,
                expiryDate: endStr
            });

            return jsonResponse({
                success: true,
                message: 'Trial Registration Successful.',
                data: {
                    status: 'REGISTERED',
                    startDate: startStr,
                    expiryDate: endStr,
                    dataSize: row[8] || 50
                }
            });
        }
    }
    trackFailedAttempt(ss, params.machineId, params.email, "TRIAL_AUTH_FAIL");
    return jsonResponse({ success: false, message: 'Authorisation Fail: Primary Credentials mismatched.' });
}

/**
 * Handles License Activation (Self-Binding Logic)
 */
function handleLicenseActivation(ss, params) {
    const sheet = ss.getSheetByName('BPP_APP_License');
    if (!sheet) return jsonResponse({ success: false, message: 'Critical Error: Sheet Missing.' });

    let data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
        const row = data[i];
        if (String(row[0]).trim() === params.licenseKey && 
            String(row[2]).trim().toLowerCase() === params.userID.trim().toLowerCase()) {

            // 1. HARDWARE BINDING logic
            const rowMachine = String(row[7]).trim().toLowerCase();
            if (!rowMachine || rowMachine === "") {
                sheet.getRange(i + 1, 8).setValue(params.machineId);
            } else if (rowMachine !== params.machineId.trim().toLowerCase()) {
                return jsonResponse({ success: false, message: 'Violation: License locked to another machine.' });
            }

            if (isExpired(String(row[6]))) return jsonResponse({ success: false, message: "License Expired." });

            const now = new Date();
            const startStr = Utilities.formatDate(now, "GMT+5:30", "dd-MM-yyyy");
            sheet.getRange(i + 1, 6).setValue(startStr);
            sheet.getRange(i + 1, 9).setValue('ACTIVATED');

            // Notify Dev
            sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "License Activation: " + params.userID, "DEV_LICENSE_SUCCESS", {
                licenseKey: params.licenseKey,
                userID: params.userID,
                machineId: params.machineId,
                expiryDate: row[6]
            });

            return jsonResponse({
                success: true,
                message: 'Activation Successful.',
                data: { status: 'ACTIVATED', expiryDate: row[6], dataSize: row[9] || 5000 }
            });
        }
    }
    return jsonResponse({ success: false, message: 'Activation Error: Key/ID invalid.' });
}

/**
 * Tracking & Audit Logic
 */
function handleTrackLogin(ss, params) {
    const now = new Date();
    const timestampStr = Utilities.formatDate(now, "GMT+5:30", "dd-MM-yyyy HH:mm:ss");

    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        let sheet = ss.getSheetByName(name);
        if (!sheet) return;
        let data = sheet.getDataRange().getValues();
        let emailCol = name.includes('Trial') ? 3 : 4;
        let machineCol = name.includes('Trial') ? 5 : 7;

        for (let i = 1; i < data.length; i++) {
            if (String(data[i][emailCol]).trim().toLowerCase() === params.email?.trim().toLowerCase() &&
                String(data[i][machineCol]).trim().toLowerCase() === params.machineId?.trim().toLowerCase()) {
                let count = (parseInt(data[i][10]) || 0) + 1;
                sheet.getRange(i + 1, 11).setValue(count); // Col K
                sheet.getRange(i + 1, 12).setValue(timestampStr); // Col L
            }
        }
    });
    return jsonResponse({ success: true });
}

function fetchLatestDeveloperMessages(ss) {
    const sheet = ss.getSheetByName('App_Config');
    if (!sheet) return jsonResponse({ success: false });
    const data = sheet.getRange(2, 3, 1, 2).getValues()[0];
    return jsonResponse({ success: true, messages: { 
        scrollNews: { message: data[0] || "", date: "" }, 
        statutory: { message: data[1] || "", date: "" } 
    }});
}

function trackFailedAttempt(ss, machineId, email, type) {
    try {
        let sheet = ss.getSheetByName('Access_Logs');
        if (!sheet) {
            sheet = ss.insertSheet('Access_Logs');
            sheet.appendRow(['Timestamp', 'Machine ID', 'Email', 'Type', 'Count']);
        }
        let data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            if (data[i][1] === machineId && data[i][2] === email) {
                let count = (data[i][4] || 0) + 1;
                sheet.getRange(i + 1, 5).setValue(count);
                if (count >= 3) {
                    sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "SECURITY ALERT", "SECURITY_ALERT", { machineId, email, type, count });
                }
                return;
            }
        }
        sheet.appendRow([new Date(), machineId, email, type, 1]);
    } catch (e) { }
}

/**
 * Helpers & Config
 */
function getLicenceFeeData(ss) {
    let sheet = ss.getSheetByName('License Fee');
    return sheet ? sheet.getDataRange().getValues() : [];
}

function getDeveloperCredentials(ss) {
    const sheet = ss.getSheetByName('Master_Config');
    if (!sheet) return { devUser: "", devPass: "" };
    return { devUser: String(sheet.getRange("A2").getValue()), devPass: String(sheet.getRange("B2").getValue()) };
}

function getAppConfig(ss) {
    const sheet = ss.getSheetByName('App_Config');
    if (!sheet) return { latestVersion: "2.0.5" };
    const d = sheet.getRange(2, 1, 1, 3).getValues()[0];
    return { latestVersion: String(d[0]), downloadUrl: String(d[1]), downloadUrlWin7: String(d[2]) };
}

function isExpired(expiryStr) {
    if (!expiryStr) return false;
    const nowStr = Utilities.formatDate(new Date(), "GMT+5:30", "yyyyMMdd");
    const p = String(expiryStr).split("-");
    if (p.length === 3) {
        return (p[2] + p[1] + p[0] < nowStr);
    }
    return false;
}

function updateExpiryStatuses(ss) {

    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {

        if (!sheet) return;
        let data = sheet.getDataRange().getValues();
        let dateCol = name.includes('Trial') ? 7 : 6;
        let statusCol = name.includes('Trial') ? 9 : 8;
        for (let i = 1; i < data.length; i++) {
            let p = String(data[i][dateCol]).split("-");
            if (p.length === 3) {
                if (p[2] + p[1] + p[0] < nowStr) {
                    sheet.getRange(i + 1, statusCol + 1).setValue("EXPIRED");
                }
            }
        }
    });
}

function updateExpiryStatusesManual() {
  updateExpiryStatuses(SpreadsheetApp.getActiveSpreadsheet());
  SpreadsheetApp.getUi().alert("🔄 Sync Complete: All expiry statuses successfully verified.");
}
