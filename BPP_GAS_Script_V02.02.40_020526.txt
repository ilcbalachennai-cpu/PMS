function fetchHistoricalMins(ss, email) {

        const targetSheet = ss.getSheetByName(shName); if (!targetSheet) return;
        const targetData = targetSheet.getDataRange().getValues();
        const emailColIdx = shName.includes('Trial') ? 3 : 5;
        for (let j = 1; j < targetData.length; j++) {

                baseMins = parseHHMMToMinutes(String(targetData[j][COL_TOTAL_HOURS - 1]));
                break;

    });
    return baseMins;
}

        const targetSheet = ss.getSheetByName(shName); if (!targetSheet) return;
        const targetData = targetSheet.getDataRange().getValues();

            if (String(targetData[j][emailColIdx]).trim().toLowerCase() === email.trim().toLowerCase()) {
                targetSheet.getRange(j+1, COL_TOTAL_HOURS).setValue(totalHHMM);
                break;
            }
        }
    });
}

function parseHHMMToMinutes(val) {
    if (val === null || val === undefined || val === "") return 0;
    
    // If it's already a number (it might be fractional days from Sheets)
    if (typeof val === 'number') {
        return Math.round(val * 24 * 60);
    }
    
    // If it's a Date object (Duration often comes as 1899-12-30 HH:mm:ss)
    if (val instanceof Date) {
        const baseDate = new Date(1899, 11, 30, 0, 0, 0);
        const diffMs = val.getTime() - baseDate.getTime();
        return Math.round(diffMs / (60 * 1000));
    }

    const str = String(val).trim();
    const match = str.match(/(\d+):(\d+)/);
    if (match) {
        let hours = parseInt(match[1]) || 0;
        const mins = parseInt(match[2]) || 0;
        return hours * 60 + mins;
    }
    
    return 0;
}

function convertMinutesToHHMM(mins) {
    const isNegative = mins < 0;
    const absMins = Math.abs(mins);
    const h = Math.floor(absMins / 60);
    const m = absMins % 60;
    return (isNegative ? "-" : "") + h + ":" + (m < 10 ? "0" + m : m);
}

function cleanupActiveUsers() {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('Active_Users');
    if (!sheet) return;

    const data = sheet.getDataRange().getValues();
    const now = new Date();

    for (let i = 1; i < data.length; i++) {
        const lastActiveStr = data[i][4];
        if (!lastActiveStr) continue;

        try {
            const parts = lastActiveStr.split(" ");
            const dateParts = parts[0].split("-");
            const timeParts = parts[1].split(":");
            const lastActiveDate = new Date(dateParts[2], dateParts[1] - 1, dateParts[0], timeParts[0], timeParts[1], timeParts[2]);

            const diffMinutes = (now.getTime() - lastActiveDate.getTime()) / (1000 * 60);
            
            if (diffMinutes > 10) {
                sheet.getRange(i + 1, 4).setValue("OFFLINE");
                sheet.getRange(i + 1, 4).setBackground('#fee2e2').setFontColor('#991b1b');
            }
        } catch (e) {
            // Skip invalid dates
        }
    }
}

            logSecurityAudit(ss, "STARTUP_VALIDATION", "VIOLATION", "Security Conflict: Email " + cleanParamsEmail + " targeted " + victimName, params);
            if (lock.hasLock()) lock.releaseLock();
            return jsonResponse({ success: false, message: "Security Conflict." });
        }
        
        logSecurityAudit(ss, "STARTUP_VALIDATION", "UNREGISTERED", "New/unregistered device — show System Initialization", params);
        if (lock.hasLock()) lock.releaseLock();
        return jsonResponse({ success: false, message: "Unregistered Device." });

    } catch (e) {
        if (lock.hasLock()) lock.releaseLock();
        logSecurityAudit(ss, "STARTUP_VALIDATION", "CRITICAL_ERROR", e.toString(), params);
        return jsonResponse({ success: false, message: "Validation Error: " + e.toString() });
    }
}

  const col = range.getColumn();
  const row = range.getRow();

  const criticalColumns = [2, 3, 5, 7, 8];

    sheet.getRange(2, 6).setValue(timestamp); // Update Column F

  }
}
