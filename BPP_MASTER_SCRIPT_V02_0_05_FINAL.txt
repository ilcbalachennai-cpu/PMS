/**
 * Handle POST requests from Electron/BPP App
 */
function doPost(e) {
    try {
        if (!e || !e.postData || !e.postData.contents) {
            return jsonResponse({ success: false, message: 'Invalid Request: No payload received.' });
        }

        const params = JSON.parse(e.postData.contents);
        const action = params.action;
        const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

        // 1. Auto-update EXPIRED status based on current date
        updateExpiryStatuses(ss);

        // 2. Route Actions
        if (action === 'REGISTER_TRIAL') {
            return handleTrialRegistration(ss, params);
        } else if (action === 'ACTIVATE_LICENSE') {
            return handleLicenseActivation(ss, params);
        } else if (action === 'VALIDATE_STARTUP') {
            return handleStartupValidation(ss, params);
        } else if (action === 'GET_MESSAGES') {
            return fetchLatestDeveloperMessages(ss);
        } else if (action === 'TRACK_LOGIN') {
            return handleTrackLogin(ss, params);
        } else if (action === 'GET_PRICING') {
            var pricingData = getLicenceFeeData(ss);
            return ContentService.createTextOutput(
                JSON.stringify({ success: true, pricing: pricingData })
            ).setMimeType(ContentService.MimeType.JSON);
        } else {
            return ContentService.createTextOutput(
                JSON.stringify({ success: false, message: 'Unknown action requested.' })
            ).setMimeType(ContentService.MimeType.JSON);
        }
    } catch (err) {
        return jsonResponse({ success: false, message: 'Server Processing Error: ' + err.toString() });
    }
}

/**
 * [ADMIN] Process License Generation
 */
function processLicenseIssuance(d) {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getSheetByName('BPP_APP_License'); 
    if (!sheet) { SpreadsheetApp.getUi().alert("Error: Sheet 'BPP_APP_License' not found!"); return; }

    var key = "BPP-" + Math.random().toString(36).substring(2, 6).toUpperCase() + "-" +
        Math.random().toString(36).substring(2, 6).toUpperCase() + "-" +
        Math.random().toString(36).substring(2, 6).toUpperCase();

    // Appended record with status PENDING
    sheet.appendRow([key, d.name, d.uid, d.mobile, d.email, "", d.expiry, "", "PENDING", d.limit]);

    sendEmailNotification(d.email, "Your BharatPay Pro License is Ready!", "LICENSE_ISSUANCE", {
        userName: d.name,
        userID: d.uid,
        licenseKey: key,
        expiryDate: d.expiry
    });

    SpreadsheetApp.getUi().alert("✅ Success!\n\nEmail sent to: " + d.email + "\nKey: " + key);
}

/**
 * Startup Logic (Fidelity Audit: Uses Col M/N for Admin Credentials)
 */
function handleStartupValidation(ss, params) {
    const config = getAppConfig(ss);
    const devCreds = getDeveloperCredentials(ss);

    // 1. Check BPP_APP_License
    let sheet = ss.getSheetByName('BPP_APP_License');
    if (sheet) {
        let data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            const row = data[i];
            const rowUserID = String(row[2]).trim().toLowerCase();
            const rowEmail = String(row[4]).trim().toLowerCase();
            const rowMachineId = String(row[7]).trim().toLowerCase();

            if (rowUserID === params.userID?.trim().toLowerCase() && rowEmail === params.email?.trim().toLowerCase() && rowMachineId === params.machineId?.trim().toLowerCase()) {
                if (isExpired(String(row[6]))) return jsonResponse({ success: false, message: "License Expired. Renew at ilcbala.BharatPayRoll@gmail.com" });

                return jsonResponse({
                    success: true,
                    message: "License Validated",
                    data: {
                        dataSize: row[9] || 5000,
                        latestVersion: config.latestVersion,
                        downloadUrl: config.downloadUrl,
                        downloadUrlWin7: config.downloadUrlWin7,
                        adminUser: row[12] || row[1], // User defined Admin Login (Col M)
                        adminPass: row[13] || "admin123", // User defined Admin Pass (Col N)
                        devUser: devCreds.devUser,
                        devPass: devCreds.devPass,
                        licenceFee: getLicenceFeeData(ss)
                    }
                });
            }
        }
    }

    // 2. Check Approved_Trial_Data
    sheet = ss.getSheetByName('Approved_Trial_Data');
    if (sheet) {
        let data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            const row = data[i];
            if (String(row[1]).trim().toLowerCase() === params.userID?.trim().toLowerCase() && String(row[3]).trim().toLowerCase() === params.email?.trim().toLowerCase() && String(row[5]).trim().toLowerCase() === params.machineId?.trim().toLowerCase()) {
                if (isExpired(String(row[7]))) return jsonResponse({ success: false, message: "Trial Expired." });

                return jsonResponse({
                    success: true,
                    message: "Trial Validated",
                    data: {
                        dataSize: row[8] || 50,
                        latestVersion: config.latestVersion,
                        downloadUrl: config.downloadUrl,
                        downloadUrlWin7: config.downloadUrlWin7,
                        adminUser: "admin",
                        adminPass: row[2] || "admin123",
                        devUser: devCreds.devUser,
                        devPass: devCreds.devPass
                    }
                });
            }
        }
    }

    trackFailedAttempt(ss, params.machineId, params.email, "STARTUP_VIOLATION");
    return jsonResponse({ success: false, message: "Unauthorised Credentials / Machine ID mismatch." });
}

/**
 * Handle Registration with Single-Row Enforcement (V02.0.05 Restoration Logic)
 */
function handleTrialRegistration(ss, params) {
    const sheet = ss.getSheetByName('Approved_Trial_Data');
    if (!sheet) return jsonResponse({ success: false, message: 'Sheet Error.' });

    let data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
        const row = data[i];
        if (String(row[3]).trim().toLowerCase() === params.email.trim().toLowerCase() && String(row[4]).trim() === String(params.mobile).trim()) {
            
            // hardware lock violation
            if (row[5] && String(row[5]).trim().toLowerCase() !== params.machineId.trim().toLowerCase()) {
                return jsonResponse({ success: false, message: 'Violation: Registered to another machine ID.' });
            }

            // Restoration logic: If status is 'REGISTERED', don't overwrite dates. Returning historical data instead.
            if (String(row[9]).trim().toUpperCase() === 'REGISTERED' && row[6] && row[7]) {
                return jsonResponse({
                    success: true,
                    message: '🛡️ History Found: Resuming your original 90-day trial for this machine.',
                    data: { status: 'REGISTERED', startDate: String(row[6]), expiryDate: String(row[7]), dataSize: row[8] || 50 }
                });
            }

            // Fresh Activation
            const now = new Date();
            const startStr = Utilities.formatDate(now, "GMT+5:30", "dd-MM-yyyy");
            const endStr = Utilities.formatDate(new Date(now.getTime() + (90 * 24 * 60 * 60 * 1000)), "GMT+5:30", "dd-MM-yyyy");

            sheet.getRange(i + 1, 1).setValue(params.userName);
            sheet.getRange(i + 1, 2).setValue(params.userID);
            sheet.getRange(i + 1, 6).setValue(params.machineId);
            sheet.getRange(i + 1, 7).setValue(startStr);
            sheet.getRange(i + 1, 8).setValue(endStr);
            sheet.getRange(i + 1, 10).setValue('REGISTERED');

            sendEmailNotification(params.email, "Trial Activated: Welcome to BharatPay Pro", "TRIAL_SUCCESS", { expiryDate: endStr });
            return jsonResponse({ success: true, message: 'Successful', data: { status: 'REGISTERED', startDate: startStr, expiryDate: endStr, dataSize: row[8] || 50 }});
        }
    }
    return jsonResponse({ success: false, message: 'Authorisation Fail: Primary Credentials mismatched.' });
}

/**
 * License Activation (Hardware Lock Binding)
 */
function handleLicenseActivation(ss, params) {
    const sheet = ss.getSheetByName('BPP_APP_License');
    if (!sheet) return jsonResponse({ success: false, message: 'Sheet missing.' });

    let data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
        const row = data[i];
        if (String(row[0]).trim() === params.licenseKey && String(row[2]).trim().toLowerCase() === params.userID.trim().toLowerCase()) {
            
            const rowMachine = String(row[7]).trim().toLowerCase();
            if (!rowMachine || rowMachine === "") {
                sheet.getRange(i + 1, 8).setValue(params.machineId); // Dynamic Binding
            } else if (rowMachine !== params.machineId.trim().toLowerCase()) {
                return jsonResponse({ success: false, message: 'Violation: Locked to another machine ID.' });
            }

            if (isExpired(String(row[6]))) return jsonResponse({ success: false, message: "Access Expired." });

            const today = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy");
            sheet.getRange(i + 1, 6).setValue(today);
            sheet.getRange(i + 1, 9).setValue('ACTIVATED');

            sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "New License Bound: " + params.userID, "DEV_LICENSE_SUCCESS", { licenseKey: params.licenseKey, userID: params.userID, machineId: params.machineId, expiryDate: row[6] });

            return jsonResponse({ success: true, message: 'Activated', data: { status: 'ACTIVATED', expiryDate: row[6], dataSize: row[9] || 5000 }});
        }
    }
    return jsonResponse({ success: false, message: 'Activation Error: Key/ID invalid.' });
}

/**
 * Tracking logic for audit (Col K/L)
 */
function handleTrackLogin(ss, params) {
    if (!params.email || !params.machineId) return jsonResponse({ success: false });
    const nowStr = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy HH:mm:ss");

    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        let sheet = ss.getSheetByName(name);
        if (!sheet) return;
        let data = sheet.getDataRange().getValues();
        let emailCol = name.includes('Trial') ? 3 : 4;
        let machineCol = name.includes('Trial') ? 5 : 7;
        for (let i = 1; i < data.length; i++) {
            if (String(data[i][emailCol]).trim().toLowerCase() === params.email?.trim().toLowerCase() && 
                String(data[i][machineCol]).trim().toLowerCase() === params.machineId?.trim().toLowerCase()) {
                let count = (parseInt(data[i][10]) || 0) + 1; // Col K
                sheet.getRange(i + 1, 11).setValue(count);
                sheet.getRange(i + 1, 12).setValue(nowStr); // Col L
            }
        }
    });
    return jsonResponse({ success: true });
}

/**
 * Helpers & Config Logic
 */
function fetchLatestDeveloperMessages(ss) {
    const sheet = ss.getSheetByName('App_Config');
    if (!sheet) return jsonResponse({ success: false });
    const data = sheet.getRange(2, 3, 1, 2).getValues()[0];
    return jsonResponse({ success: true, messages: { scrollNews: { message: data[0] || "", date: "" }, statutory: { message: data[1] || "", date: "" } } });
}

function getAppConfig(ss) {
    const sheet = ss.getSheetByName('App_Config');
    if (!sheet) return { latestVersion: "2.0.5", downloadUrl: "", downloadUrlWin7: "" };
    const d = sheet.getRange(2, 1, 1, 3).getValues()[0];
    return { latestVersion: String(d[0] || "2.0.5"), downloadUrl: String(d[1] || ""), downloadUrlWin7: String(d[2] || "") };
}

function getDeveloperCredentials(ss) {
    const sheet = ss.getSheetByName('Master_Config');
    if (!sheet) return { devUser: "", devPass: "" };
    return { devUser: String(sheet.getRange("A2").getValue() || ""), devPass: String(sheet.getRange("B2").getValue() || "") };
}

function getLicenceFeeData(ss) {
    let sheet = ss.getSheetByName('License Fee');
    return sheet ? sheet.getDataRange().getValues() : [];
}

function isExpired(expiryStr) {
    if (!expiryStr) return false;
    const nowStr = Utilities.formatDate(new Date(), "GMT+5:30", "yyyyMMdd");
    const p = String(expiryStr).split("-");
    return p.length === 3 ? (p[2] + p[1] + p[0] < nowStr) : false;
}

function updateExpiryStatuses(ss) {
    const nowStr = Utilities.formatDate(new Date(), "GMT+5:30", "yyyyMMdd");
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        let sheet = ss.getSheetByName(name);
        if (!sheet) return;
        let data = sheet.getDataRange().getValues();
        let dateCol = name.includes('Trial') ? 7 : 6;
        let statusCol = name.includes('Trial') ? 9 : 8;
        for (let i = 1; i < data.length; i++) {
            let p = String(data[i][dateCol]).split("-");
            if (p.length === 3 && (p[2] + p[1] + p[0] < nowStr)) {
                sheet.getRange(i + 1, statusCol + 1).setValue("EXPIRED");
            }
        }
    });
}

function updateExpiryStatusesManual() {
  updateExpiryStatuses(SpreadsheetApp.getActiveSpreadsheet());
  SpreadsheetApp.getUi().alert("🔄 Sync Complete!");
}

function jsonResponse(obj) {
    return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON);
}
