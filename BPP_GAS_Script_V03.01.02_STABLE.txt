const SPREADSHEET_ID = SpreadsheetApp.getActiveSpreadsheet().getId();
const COL_MACHINE_ID = 9; // Column I
const COL_STATUS = 10;    // Column J
const COL_LOGIN_COUNT = 11; // Column K
const COL_LAST_LOGIN = 12;  // Column L
const COL_TOTAL_HOURS = 13; // Column M

function doGet(e) { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const params = e.parameter.action ? e.parameter : JSON.parse(e.postData.contents);
  const action = params.action;

  try {
    switch (action) {
      case 'REGISTER_TRIAL': return handleTrialRegistration(ss, params);
      case 'ACTIVATE_LICENSE': return handleLicenseActivation(ss, params);
      case 'TRACK_LOGIN': return handleTrackLogin(ss, params);
      case 'GET_MESSAGES': return fetchLatestDeveloperMessages(ss);
      case 'UPDATE_MESSAGES': return handleUpdateMessages(ss, params);
      case 'HEARTBEAT': return handleTrackHeartbeat(ss, params);
      case 'VERIFY_EMAIL': return handleVerifyIdentityEmail(ss, params);
      case 'SYNC_REPAIR': return handleSyncIdentityRepair(ss, params);
      case 'REQUEST_RESET_OTP': return handleRequestOTPReset(ss, params);
      case 'VERIFY_RESET_OTP': return handleVerifyResetOTP(ss, params);
      case 'UPDATE_PASSWORD': return handlePasswordUpdate(ss, params);
      case 'REQUEST_DEV_OTP': return handleRequestDevOTP(ss, params);
      case 'VERIFY_DEV_OTP': return handleVerifyDevOTP(ss, params);
      case 'REQUEST_RESTORE_OTP': return handleRequestRestoreOTP(ss, params);
      case 'VERIFY_RESTORE_OTP': return handleVerifyRestoreOTP(ss, params);
      case 'SEND_REG_OTP': return handleSendRegOTP(ss, params);
      case 'VERIFY_REG_OTP': return handleVerifyRegOTP(ss, params);
      default: return jsonResponse({ success: false, message: "Unknown action: " + action });
    }
  } catch (err) {
    return jsonResponse({ success: false, message: err.toString() });
  }
}

function handleTrialRegistration(ss, params) {

        const idRegex = /^[A-Z0-9]{5,12}$/;
        if (!params.userID || !idRegex.test(params.userID)) {
            if (lock.hasLock()) lock.releaseLock();
            return jsonResponse({ success: false, message: "Invalid User ID format (5-12 Alphanumeric characters required)." });
        }
        const sheet = ss.getSheetByName('Approved_Trial_Data'); 
        if (!sheet) { if (lock.hasLock()) lock.releaseLock(); return jsonResponse({ success: false, message: "Trial database not found." }); }
        
        const data = sheet.getDataRange().getValues();
        const today = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy");
        
        for (let i = 1; i < data.length; i++) {
            if (String(data[i][3]).toLowerCase() === params.email.toLowerCase()) {
                const expDate = new Date(); expDate.setDate(expDate.getDate() + 90);
                const expStr  = Utilities.formatDate(expDate, "GMT+5:30", "dd-MM-yyyy");
                
                const existingPass = String(data[i][2]).trim();
                const finalPass = params.password || existingPass || generateHardenedPassword();
                const finalUserID = String(params.userID || "").toUpperCase().trim();
                
                // V03.01.02 FIX: Define isSyncRegistration properly
                const isSyncRegistration = !!(existingPass && existingPass !== "PENDING" && existingPass !== "");

                sheet.getRange(i+1, 1, 1, 12).setValues([[
                    params.userName.toUpperCase(), finalUserID, finalPass, data[i][3], data[i][4], 
                    params.machineId, today, expStr, data[i][8], 'TRIAL ACTIVE', 0, today
                ]]);
                
                const applicantEmail = String(params.email || "").trim();
                const applicantName  = String(params.userName || "User").trim();
                const devEmail       = "ilcbala.BharatPayRoll@gmail.com";

                if (isSyncRegistration) {
                    sendEmailNotification(applicantEmail, "Credentials Synchronized", "CREDENTIALS_SYNCED", { userName: applicantName });
                } else {
                    const welcomeHtml = buildProfessionalEmail(
                        "Welcome to BharatPay Pro",

function handleLicenseActivation(ss, params) {
    const lock = LockService.getScriptLock();
    try {
        lock.waitLock(10000);
        const sheet = ss.getSheetByName('BPP_APP_License');
        if (!sheet) return jsonResponse({ success: false, message: "License database not found." });
        const data = sheet.getDataRange().getValues();
        
        for (let i = 1; i < data.length; i++) {
            const keyMatch = String(data[i][0]).replace(/-/g,"") === params.licenseKey.replace(/-/g,"");
            const idMatch  = String(data[i][2]).toLowerCase() === String(params.userID || "").toLowerCase();
            
            if (keyMatch && idMatch) {
                if (params.appPassword) sheet.getRange(i+1, 4).setValue(params.appPassword);

                const existingStartDate = String(data[i][6]).trim();
                const isSyncActivation = !!(existingStartDate && existingStartDate !== "PENDING" && existingStartDate !== "");
                
                if (!isSyncActivation) sheet.getRange(i+1, 7).setValue(Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy"));
                
                const secureMachineId = (params.machineId && params.machineId.toUpperCase() !== "EXPIRED") ? params.machineId : "";
                if (secureMachineId) sheet.getRange(i+1, COL_MACHINE_ID).setValue(secureMachineId);
                
                sheet.getRange(i+1, COL_STATUS).setValue('LICENSE ACTIVE');
                
                if (lock.hasLock()) lock.releaseLock();
                return jsonResponse({ success: true, data: { status: 'LICENSE ACTIVE', expiryDate: data[i][7], dataSize: data[i][10] || 5000, password: data[i][3] } });
            }
        }
        if (lock.hasLock()) lock.releaseLock();
        return jsonResponse({ success: false, message: "Invalid Credentials." });
    } catch (e) {
        if (lock.hasLock()) lock.releaseLock();
        return jsonResponse({ success: false, message: e.toString() });
    }
}

function handleTrackLogin(ss, params) {
    const now = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy HH:mm:ss");
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        const sheet = ss.getSheetByName(name); if (!sheet) return;
        const isTrial = name.includes('Trial');
        const eCol = isTrial ? 3 : 5;
        const mCol = isTrial ? 5 : 8;
        const data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            if (String(data[i][eCol]).toLowerCase().trim() === params.email.toLowerCase().trim()) {
                const countCell = sheet.getRange(i+1, COL_LOGIN_COUNT);
                countCell.setValue((parseInt(data[i][COL_LOGIN_COUNT-1]) || 0) + 1);
                sheet.getRange(i+1, COL_LAST_LOGIN).setValue(now);
                break;
            }
        }
    });
    return jsonResponse({ success: true });
}

function fetchLatestDeveloperMessages(ss) {
    const config = getAppConfig(ss);
    const sheet = ss.getSheetByName('Developer_Message');
    let messages = { scrollNews: { message: "", date: "" }, statutory: { message: "", date: "", header: "ANNOUNCEMENT" }, flashPopup: { message: "", date: "", header: "FLASH ALERT" } };
    if (sheet) {
        const data = sheet.getDataRange().getValues();
        for (let i = data.length - 1; i >= 1; i--) {
            const type = String(data[i][3]).toUpperCase();
            if (type === "NEWS" && !messages.scrollNews.message) messages.scrollNews = { message: data[i][2], date: data[i][1] };
            else if (type === "MESSAGE" && !messages.statutory.message) messages.statutory = { message: data[i][2], date: data[i][1], header: data[i][5] };
            else if (type === "FLASH" && !messages.flashPopup.message) messages.flashPopup = { message: data[i][2], date: data[i][1], header: data[i][5] };
        }
    }
    return jsonResponse({ success: true, messages: messages, latestVersion: config.latestVersion, downloadUrl: config.downloadUrl, downloadUrlWin7: config.downloadUrlWin7, launcherUrl: config.launcherUrl, patchTimestamp: config.patchTimestamp });
}

function getAppConfig(ss) {
    const sheet = ss.getSheetByName('App_Config');
    if (!sheet) return { latestVersion: "03.01.02", launcherUrl: "" };
    const d = sheet.getRange(2, 1, 1, 8).getValues()[0];
    return { latestVersion: String(d[0] || "03.01.02"), downloadUrl: String(d[1] || ""), downloadUrlWin7: String(d[2] || ""), launcherUrl: String(d[3] || ""), patchTimestamp: String(d[5] || "") };
}

function generateHardenedPassword() { return "BPP" + Math.floor(1000 + Math.random() * 9000); }
function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); }

      <div style="padding:30px;"><p><strong>${greeting}</strong>,</p>${contentHtml}</div>

}

function sendEmailNotification(to, subject, type, data) {
    const html = buildProfessionalEmail(subject, "Security Alert", "#1e40af", "Hello " + (data.userName || "User"), `<p>Status Sync Event: ${type}</p>`);
    try { MailApp.sendEmail({ to: to, subject: subject, htmlBody: html }); } catch (e) {}
}
