function getAuthSecret() {
    const cache = CacheService.getScriptCache();
    let secret = cache.get("BPP_AUTH_SECRET");
    if (!secret) {
        secret = PropertiesService.getScriptProperties().getProperty("BPP_AUTH_SECRET") || "BPP-ULTIMATE-V2-SECURE";
        cache.put("BPP_AUTH_SECRET", secret, 21600); // Cache for 6 hours
    }
    return secret;
}

function doGet(e) {

function doPost(e) {
    try {
        if (!e || !e.postData || !e.postData.contents) return jsonResponse({ success: false, message: 'No payload.' });
        const params = JSON.parse(e.postData.contents);
        const action = params.action;
        const ss     = SpreadsheetApp.openById(SPREADSHEET_ID);

        // --- V02.02.21: SELECTIVE VERSION SENTRY (Hard Guard) ---
        const SUPPORT_FLOOR = "02.02.13"; // Versions below this are formally sunsetted
        const currentVersion = params.version || "02.02.12";
        
        if (compareVersions(currentVersion, SUPPORT_FLOOR) < 0) {
            const config = getAppConfig(ss);
            return jsonResponse({ 
                success: false, 
                status: "BLOCK_LEGACY",
                launcherUrl: config.launcherUrl,
                message: "Critical Update Required: Your current version (" + currentVersion + ") is no longer supported. Please download the latest launcher to refresh your application." 
            });
        }

        // --- V02.02.21: CLOUD GATEKEEPER (Authorization) ---
        if (params.authSecret !== getAuthSecret()) {
            logSecurityAudit(ss, "DO_POST", "UNAUTHORIZED", "Invalid Auth Secret provided from: " + (params.userID || "Unknown"), params);
            return jsonResponse({ success: false, message: "Security Violation: Unauthorized Cloud Access rejected." });
        }

        // --- DEVELOPER BYPASS (Isolated Auth - No Sheet Data Updates) ---
        if (action === 'REQUEST_DEV_OTP')  return handleRequestDevOTP(ss, params);
        if (action === 'VERIFY_DEV_OTP')   return handleVerifyDevOTP(ss, params);

        updateExpiryStatuses(ss);

        if (action === 'VALIDATE_STARTUP')  return handleStartupValidation(ss, params);
        if (action === 'REGISTER_TRIAL')    return handleTrialRegistration(ss, params);
        if (action === 'ACTIVATE_LICENSE')  return handleLicenseActivation(ss, params);
        if (action === 'UPDATE_PASSWORD')   return handlePasswordUpdate(ss, params);
        if (action === 'REQUEST_OTP_RESET') return handleRequestOTPReset(ss, params);
        if (action === 'GET_MESSAGES')      return fetchLatestDeveloperMessages(ss);
        if (action === 'UPDATE_MESSAGES')   return handleUpdateMessages(ss, params);
        if (action === 'TRACK_LOGIN')       return handleTrackLogin(ss, params);
        if (action === 'HEARTBEAT')         return handleTrackHeartbeat(ss, params);
        if (action === 'GET_PRICING')       return jsonResponse({ success: true, pricing: getLicenceFeeData(ss) });
        if (action === 'VERIFY_IDENTITY_EMAIL') return handleVerifyIdentityEmail(ss, params);
        if (action === 'SYNC_IDENTITY_REPAIR')  return handleSyncIdentityRepair(ss, params);
        if (action === 'REQUEST_RESTORE_OTP')   return handleRequestRestoreOTP(ss, params);

    }

function fetchHistoricalMins(ss, email) {
    let baseMins = 0;
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(function(shName) {
        if (baseMins > 0) return;
        const targetSheet = ss.getSheetByName(shName); if (!targetSheet) return;
        const targetData = targetSheet.getDataRange().getValues();
        const emailColIdx = shName.includes('Trial') ? 3 : 5;
        for (let j = 1; j < targetData.length; j++) {
            if (String(targetData[j][emailColIdx]).trim().toLowerCase() === email.trim().toLowerCase()) {
                baseMins = parseHHMMToMinutes(String(targetData[j][COL_TOTAL_HOURS - 1]));
                break;
            }
        }
    });
    return baseMins;
}

function syncCumulativeHours(ss, email, baseMins, sessH, sessM) {
    const totalMins = baseMins + (sessH * 60) + sessM;
    const totalHHMM = convertMinutesToHHMM(totalMins);
    
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(function(shName) {
        const targetSheet = ss.getSheetByName(shName); if (!targetSheet) return;
        const targetData = targetSheet.getDataRange().getValues();
        const emailColIdx = shName.includes('Trial') ? 3 : 5;
        for (let j = 1; j < targetData.length; j++) {
            if (String(targetData[j][emailColIdx]).trim().toLowerCase() === email.trim().toLowerCase()) {
                targetSheet.getRange(j+1, COL_TOTAL_HOURS).setValue(totalHHMM);
                break;
            }
        }
    });
}

function parseHHMMToMinutes(val) {
    if (val === null || val === undefined || val === "") return 0;
    
    // If it's already a number (it might be fractional days from Sheets)
    if (typeof val === 'number') {
        return Math.round(val * 24 * 60);
    }
    
    // If it's a Date object (Duration often comes as 1899-12-30 HH:mm:ss)
    if (val instanceof Date) {
        const baseDate = new Date(1899, 11, 30, 0, 0, 0);
        const diffMs = val.getTime() - baseDate.getTime();
        return Math.round(diffMs / (60 * 1000));
    }

    const str = String(val).trim();
    const match = str.match(/(\d+):(\d+)/);
    if (match) {
        let hours = parseInt(match[1]) || 0;
        const mins = parseInt(match[2]) || 0;
        return hours * 60 + mins;
    }
    
    return 0;
}

function convertMinutesToHHMM(mins) {
    const isNegative = mins < 0;
    const absMins = Math.abs(mins);
    const h = Math.floor(absMins / 60);
    const m = absMins % 60;
    return (isNegative ? "-" : "") + h + ":" + (m < 10 ? "0" + m : m);
}

function cleanupActiveUsers() {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    const sheet = ss.getSheetByName('Active_Users');
    if (!sheet) return;

    const data = sheet.getDataRange().getValues();
    const now = new Date();

    for (let i = 1; i < data.length; i++) {
        const lastActiveStr = data[i][4];
        if (!lastActiveStr) continue;

        try {
            const parts = lastActiveStr.split(" ");
            const dateParts = parts[0].split("-");
            const timeParts = parts[1].split(":");
            const lastActiveDate = new Date(dateParts[2], dateParts[1] - 1, dateParts[0], timeParts[0], timeParts[1], timeParts[2]);

            const diffMinutes = (now.getTime() - lastActiveDate.getTime()) / (1000 * 60);
            
            if (diffMinutes > 10) {
                sheet.getRange(i + 1, 4).setValue("OFFLINE");
                sheet.getRange(i + 1, 4).setBackground('#fee2e2').setFontColor('#991b1b');
            }
        } catch (e) {
            // Skip invalid dates
        }
    }
}

function handleStartupValidation(ss, params) {
    const lock = LockService.getScriptLock();
    try {
        lock.waitLock(10000); // Wait 10s for other threads
        const config   = getAppConfig(ss);
        const devCreds = getDeveloperCredentials(ss);
        
        // Helper to normalize mobile numbers (remove non-digits, last 10 digits)
        const normalizeMobile = (m) => {
            const digits = String(m || "").replace(/\D/g, "");
            return digits.length > 10 ? digits.slice(-10) : digits;
        };

        let conflictFound       = false;
        let conflictTargetEmail = "";
        let victimName          = "";

        const cleanParamsID      = String(params.userID    || "").trim();
        const cleanParamsEmail   = String(params.email     || "").trim().toLowerCase();
        const cleanParamsMachine = String(params.machineId || "").trim().toLowerCase();
        const cleanParamsMobile  = String(params.mobile    || "").trim();

        const isDevMachine = !!(devCreds.devMachine && cleanParamsMachine === devCreds.devMachine.trim().toLowerCase());
        
        // --- V02.02.15: ISOLATED DEVELOPER STARTUP BYPASS (Refined) ---
        const isExplicitDevLogin = (cleanParamsID === devCreds.devUser) || (cleanParamsID === "" && isDevMachine);
        
        if (isDevMachine && isExplicitDevLogin) {
            if (lock.hasLock()) lock.releaseLock();
            return jsonResponse({
                success: true, message: "Developer Environment Authorized",
                data: { 
                    isTrial: false, status: 'DEVELOPER ACTIVE', userName: "Developer", userID: devCreds.devUser, 
                    latestVersion: config.latestVersion, adminUser: devCreds.devUser, adminPass: devCreds.devPass, 
                    devUser: devCreds.devUser, devPass: devCreds.devPass, isDeveloper: true 
                }
            });
        }

        const isRescueMode = (cleanParamsEmail === "rescue" || !cleanParamsEmail) && cleanParamsID.length > 3;

        // --- PHASE 1: SEARCH LICENSED DATABASE (Enterprise Priorities) ---
        let licenseSheet = ss.getSheetByName('BPP_APP_License');
        let userFoundInLicense = false;

        if (licenseSheet) {
            const data = licenseSheet.getDataRange().getValues();
            for (let i = 1; i < data.length; i++) {
                const row          = data[i];
                const rowKey       = String(row[0]).replace(/-/g,"").trim().toLowerCase();
                const rowID_ColC   = String(row[2]).trim(); // User ID (Col C)
                const rowID_ColB   = String(row[1]).trim(); // Full Name (Col B)
                const rowPass      = String(row[3]).trim(); // Password (Col D)
                const rowMobile    = String(row[4]).trim(); // Mobile (Col E)
                const rowEmail     = String(row[5]).trim().toLowerCase(); // Email (Col F)
                const rowMachineId = String(row[8]).trim().toLowerCase(); // Machine ID (Col I)
                const rowExpiry    = String(row[7]);

                // --- V02.02.21: STRICT 4-FIELD SYNC VALIDATION ---
                if (rowMachineId && rowMachineId === cleanParamsMachine) {
                    userFoundInLicense = true;
                    
                    const idMatch   = (rowID_ColC.toLowerCase() === cleanParamsID.toLowerCase());
                    const passMatch = (rowPass === params.appPassword);
                    const emailMatch = rowEmail === cleanParamsEmail || cleanParamsEmail === "rescue";
                    
                    const cleanParamsKey = String(params.licenseKey || "").replace(/-/g,"").trim().toLowerCase();
                    const keyMatch = (cleanParamsKey === "rescue" || cleanParamsKey === rowKey || cleanParamsKey === "trial");

                    if (idMatch && passMatch && emailMatch && keyMatch && !isRescueMode) {
                        const expired = isExpired(rowExpiry);
                        if (!expired) {
                            logSecurityAudit(ss, "STARTUP_VALIDATION", "SUCCESS", "License Match (4-Field) for " + rowID_ColC, params);
                            if (lock.hasLock()) lock.releaseLock();
                            return jsonResponse({
                                success: true, message: "License Validated",
                                data: { 
                                    isTrial: false, licenseKey: row[0], userName: row[1], userID: String(rowID_ColC).toUpperCase(), registeredTo: row[5], registeredMobile: row[4], 
                                    startDate: row[6], expiryDate: row[7], status: 'LICENSE ACTIVE',
                                    dataSize: parseInt(row[10]) || 5000, 
                                    latestVersion: config.latestVersion, adminUser: String(rowID_ColC).toUpperCase(), adminPass: row[3], 
                                    devUser: devCreds.devUser, devPass: devCreds.devPass, licenceFee: getLicenceFeeData(ss) 
                                }
                            });
                        } else {
                            if (lock.hasLock()) lock.releaseLock();
                            return jsonResponse({
                                success: true, message: "LICENSE EXPIRED",
                                data: { status: 'LICENSE_EXPIRED', expiryDate: row[7], userName: row[1] }
                            });
                        }
                    } else {
                        // HARDWARE MATCHES BUT IDENTITY/CREDENTIALS MISMATCH -> TRIGGER RESTORE
                        logSecurityAudit(ss, "STARTUP_VALIDATION", "RECOVERY_TRIGGERED", "Hardware Match but Credential Mismatch for: " + rowID_ColC, params);
                        if (lock.hasLock()) lock.releaseLock();
                        return jsonResponse({
                            success: true, 
                            message: "IDENTITY_RESTORE_REQUIRED", 
                            data: { status: 'PENDING_RESTORE', userName: row[1], userID: String(rowID_ColC).toUpperCase() }
                        });
                    }
                }
                
                // Track email conflicts (same email, different machine)
                if (!userFoundInLicense && cleanParamsEmail !== "rescue" && cleanParamsEmail === rowEmail) {
                    const existingMachine = String(row[8]).trim().toLowerCase();
                    if (existingMachine && existingMachine !== cleanParamsMachine) {
                        conflictFound = true; conflictTargetEmail = rowEmail; victimName = String(row[1]);
                    }
                }
            }
        }

        // --- PHASE 2: SEARCH TRIAL DATABASE (Fallback only if no valid Full License found) ---
        if (!userFoundInLicense) {
            const trialSheet = ss.getSheetByName('Approved_Trial_Data');
            if (trialSheet) {
                const data = trialSheet.getDataRange().getValues();
                for (let i = 1; i < data.length; i++) {
                    const row          = data[i];
                    const rowName      = String(row[0]).trim(); // Name (Col A)
                    const rowID        = String(row[1]).trim(); // User ID (Col B)

                    const rowMachineId = String(row[5]).trim().toLowerCase(); // Machine ID (Col F)
                    const rowExpiry    = String(row[7]); // End Date (Col H)

                    // --- V02.02.21: TRIAL 4-FIELD SYNC VALIDATION ---
                    if (rowMachineId && rowMachineId === cleanParamsMachine) {
                        const idMatch   = rowID.toLowerCase() === cleanParamsID.toLowerCase();
                        const passMatch = (rowPass === params.appPassword);
                        const emailMatch = rowEmail === cleanParamsEmail || cleanParamsEmail === "rescue";

                        if (idMatch && passMatch && emailMatch && !isRescueMode) {
                            const expired = isExpired(rowExpiry);
                            logSecurityAudit(ss, "STARTUP_VALIDATION", "SUCCESS", "Trial Match (4-Field) for " + rowName, params);
                            if (lock.hasLock()) lock.releaseLock();
                            return jsonResponse({
                                success: true, message: "Trial Validated",
                                data: { 
                                    isTrial: true, licenseKey: "TRIAL", userName: rowName, userID: rowID.toUpperCase(), registeredTo: row[3], registeredMobile: row[4], 
                                    startDate: row[6], expiryDate: row[7], 
                                    status: expired ? 'TRIAL_EXPIRED' : 'TRIAL ACTIVE',
                                    dataSize: parseInt(row[8]) || 50, 
                                    latestVersion: config.latestVersion, adminUser: rowID.toUpperCase(), adminPass: row[2], 
                                    devUser: devCreds.devUser, devPass: devCreds.devPass 
                                }
                            });
                        } else {
                            // HARDWARE MATCHES BUT IDENTITY/CREDENTIALS MISMATCH -> TRIGGER RESTORE
                            logSecurityAudit(ss, "STARTUP_VALIDATION", "RECOVERY_TRIGGERED", "Trial Hardware Match but Credential Mismatch for: " + rowName, params);
                            if (lock.hasLock()) lock.releaseLock();
                            return jsonResponse({
                                success: true, 
                                message: "IDENTITY_RESTORE_REQUIRED", 
                                data: { status: 'PENDING_RESTORE', userName: rowName, userID: rowID }
                            });
                        }
                    } else if (cleanParamsEmail !== "rescue" && cleanParamsEmail === rowEmail) {
                        // Email conflict - used by someone else on different hardware
                        const existingMachine = String(row[5]).trim().toLowerCase();
                        if (existingMachine && existingMachine !== cleanParamsMachine) {
                            conflictFound = true;
                            conflictTargetEmail = rowEmail;
                            victimName = String(row[0]);
                        }
                    }
                }
            }
        }

        // ============================================================
        // V02.02.17 CRITICAL FIX:
        // The old code used `conflictFound || !isDevMachine` which caused
        // EVERY unregistered non-developer machine to trigger a Security
        // Violation email, blocking all legitimate users.
        //
        // NEW LOGIC:
        //  - conflictFound = TRUE  → REAL THREAT: same email on 2 machines
        //                           → Send security violation email + block
        //  - conflictFound = FALSE → UNREGISTERED DEVICE: new clean install
        //                           → Return simple "Unregistered Device"
        //                             so app shows System Initialization page
        //                             (no false security violation email)
        // ============================================================
        if (conflictFound) {
            // REAL SECURITY CONFLICT: Email is already bound to a different machine
            const securityData = { 
                userName: victimName || "System User", 
                attemptID: params.userID, 
                attemptEmail: params.email, 
                attemptMachine: params.machineId, 
                timestamp: new Date().toLocaleString() 
            };
            if (conflictTargetEmail) sendEmailNotification(conflictTargetEmail, "⚠️ Security Alert", "VIOLATION", securityData);
            sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "🚨 Security Conflict Alert", "VIOLATION", securityData);
            logSecurityAudit(ss, "STARTUP_VALIDATION", "VIOLATION", "Security Conflict: Email " + cleanParamsEmail + " targeted " + victimName, params);
            if (lock.hasLock()) lock.releaseLock();
            return jsonResponse({ success: false, message: "Security Conflict." });
        }
        
        // UNREGISTERED DEVICE (clean new install — NOT a security violation)
        logSecurityAudit(ss, "STARTUP_VALIDATION", "UNREGISTERED", "New/unregistered device — show System Initialization", params);
        if (lock.hasLock()) lock.releaseLock();
        return jsonResponse({ success: false, message: "Unregistered Device." });

    } catch (e) {
        if (lock.hasLock()) lock.releaseLock();
        logSecurityAudit(ss, "STARTUP_VALIDATION", "CRITICAL_ERROR", e.toString(), params);
        return jsonResponse({ success: false, message: "Validation Error: " + e.toString() });
    }
}

function checkLicenseConflict(email) {
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('BPP_APP_License'); if (!sheet) return { exists: false };
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) { if (String(data[i][5]).trim().toLowerCase() === email.trim().toLowerCase()) return { exists: true, name: data[i][1], row: i + 1 }; }
    return { exists: false };
}

function processLicenseIssuance(d) {
    try {
        const storedOTP = PropertiesService.getScriptProperties().getProperty('ADMIN_OTP');
        if (!storedOTP || d.otp !== storedOTP) return { success: false, message: "Invalid OTP Code." };
        PropertiesService.getScriptProperties().deleteProperty('ADMIN_OTP');
        const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName('BPP_APP_License');
        const genBlock = () => Math.random().toString(36).substring(2, 6).toUpperCase();
        const key = genBlock() + "-" + genBlock() + "-" + genBlock() + "-" + genBlock();
        
        let finalPass = generateHardenedPassword();
        let isForcedNew = true;

        if (!d.resetPass) {
            if (d.existingRow && d.existingRow !== "NEW") {
                const oldPass = sheet.getRange(parseInt(d.existingRow), 4).getValue();
                if (oldPass) { finalPass = oldPass; isForcedNew = false; }
            } else {
                const trialSheet = ss.getSheetByName('Approved_Trial_Data');
                if (trialSheet) {
                    const tData = trialSheet.getDataRange().getValues();
                    for (let j = 1; j < tData.length; j++) {
                        if (String(tData[j][3]).toLowerCase() === d.email.toLowerCase()) {
                            const trialPass = tData[j][2];
                            if (trialPass) { finalPass = trialPass; isForcedNew = false; }
                            break;
                        }
                    }
                }
            }
        }

        const issueDate = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy");
        const rowData = [key, d.name.toUpperCase(), d.uid, finalPass, d.mobile, d.email, issueDate, d.expiry, "", "PENDING", d.limit];
        if (d.existingRow && d.existingRow !== "NEW") sheet.getRange(parseInt(d.existingRow), 1, 1, 11).setValues([rowData]);
        else sheet.appendRow(rowData);
        
        sendEmailNotification(d.email, "BharatPay Pro License Details", "LICENSE_ISSUANCE", { userName: d.name, userID: d.uid, licenseKey: key, expiryDate: d.expiry, limit: d.limit, pass: finalPass, isNewPass: isForcedNew });
        return { success: true };
    } catch (e) { return { success: false, message: e.toString() }; }
}

function handleRequestOTPReset(ss, params) {
    const email = String(params.email || "").trim().toLowerCase();
    const userID = String(params.userID || "").trim().toLowerCase();
    if (!email || !userID) return jsonResponse({ success: false, message: "Missing credentials." });

    let foundUser = null;
    let userName = "User";

    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        if (foundUser) return;
        const sheet = ss.getSheetByName(name);
        if (!sheet) return;
        const data = sheet.getDataRange().getValues();
        const eCol = name.includes('Trial') ? 3 : 5;
        const uCol = name.includes('Trial') ? 1 : 2;
        const nCol = name.includes('Trial') ? 0 : 1;
        
        for (let i = 1; i < data.length; i++) {
            if (String(data[i][eCol]).toLowerCase() === email && String(data[i][uCol]).toLowerCase() === userID) {
                foundUser = email;
                userName = String(data[i][nCol]);
                break;
            }
        }
    });

    if (!foundUser) return jsonResponse({ success: false, message: "Identity not found in database." });

    const otp = Math.floor(100000 + Math.random() * 900000).toString();
    PropertiesService.getScriptProperties().setProperty('OTP_' + foundUser, otp);
    
    sendEmailNotification(foundUser, "BharatPay Pro Reset Code", "PASSWORD_RESET", { userName: userName, tempCode: otp });
    
    return jsonResponse({ success: true, message: "OTP sent to " + email });
}

function handleVerifyDevOTP(ss, params) {
    const devCreds = getDeveloperCredentials(ss);
    const otp = String(params.otp || "").trim();
    
    const storedOTP = PropertiesService.getScriptProperties().getProperty('DEV_OTP_' + devCreds.devUser);
    const storedTime = PropertiesService.getScriptProperties().getProperty('DEV_OTP_TIME_' + devCreds.devUser);
    
    if (!storedOTP || otp !== storedOTP) return jsonResponse({ success: false, message: "Invalid or expired Bypass Code." });
    
    const now = new Date().getTime();
    if (!storedTime || (now - parseInt(storedTime)) > 120000) {
        PropertiesService.getScriptProperties().deleteProperty('DEV_OTP_' + devCreds.devUser);
        PropertiesService.getScriptProperties().deleteProperty('DEV_OTP_TIME_' + devCreds.devUser);
        return jsonResponse({ success: false, message: "Bypass Code has EXPIRED (2-minute limit)." });
    }
    
    PropertiesService.getScriptProperties().deleteProperty('DEV_OTP_' + devCreds.devUser);
    PropertiesService.getScriptProperties().deleteProperty('DEV_OTP_TIME_' + devCreds.devUser);
    return jsonResponse({ 
        success: true, 
        message: "Developer Verified",
        data: {
          devUser: devCreds.devUser,
          devPass: devCreds.devPass,
          name: devCreds.devUser + " (Developer)",
          role: "Developer"
        }
    });
}

function handlePasswordUpdate(ss, params) {
    const email = String(params.email || "").trim().toLowerCase();
    const newPass = String(params.newPassword || "").trim();
    const otp = String(params.otp || "").trim();
    const machineId = String(params.machineId || "").trim();

    if (!email || !newPass || !otp) return jsonResponse({ success: false, message: "Missing data." });

    const storedOTP = PropertiesService.getScriptProperties().getProperty('OTP_' + email);
    if (!storedOTP || otp !== storedOTP) return jsonResponse({ success: false, message: "Invalid or expired OTP." });

    let updated = false;
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(name => {
        if (updated) return;
        const sheet = ss.getSheetByName(name);
        if (!sheet) return;
        const data = sheet.getDataRange().getValues();
        const eCol = name.includes('Trial') ? 3 : 5;
        const pCol = name.includes('Trial') ? 2 : 3;
        const mCol = name.includes('Trial') ? 5 : 8;

        for (let i = 1; i < data.length; i++) {
            if (String(data[i][eCol]).toLowerCase() === email) {
                const rowMachine = String(data[i][mCol]);
                if (!rowMachine || rowMachine === machineId) {
                    sheet.getRange(i+1, pCol+1).setValue(newPass);
                    updated = true;
                    break;
                }
            }
        }
    });

    if (updated) {
        PropertiesService.getScriptProperties().deleteProperty('OTP_' + email);
        return jsonResponse({ success: true, message: "Password updated successfully." });
    }

    return jsonResponse({ success: false, message: "Update failed. Account/Hardware mismatch." });
}

function handleLicenseActivation(ss, params) {
    const lock = LockService.getScriptLock();
    try {
        lock.waitLock(10000);
        const sheet = ss.getSheetByName('BPP_APP_License'); if (!sheet) return jsonResponse({ success: false, message: "License database not found." });
        const data = sheet.getDataRange().getValues();
        
        // V02.02.17 FIX: Case-insensitive userID comparison to prevent VENKY67 vs VENKY67 match failures
        for (let i = 1; i < data.length; i++) {
            const keyMatch = String(data[i][0]).replace(/-/g,"") === params.licenseKey.replace(/-/g,"");
            const idMatch  = String(data[i][2]).toLowerCase() === String(params.userID || "").toLowerCase();
            
            if (keyMatch && idMatch) {
                // --- V02.02.21: SYNC USER-SET PASSWORD DURING ACTIVATION ---
                if (params.appPassword) {
                    sheet.getRange(i+1, 4).setValue(params.appPassword); // Index 3 / Col D
                }

                // --- V02.02.21: DATE PERSISTENCE & SYNC DETECTION ---
                const existingStartDate = String(data[i][6]).trim();
                const isSyncActivation = existingStartDate && existingStartDate !== "PENDING" && existingStartDate !== "";
                
                if (!isSyncActivation) {
                    sheet.getRange(i+1, 7).setValue(Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy")); 
                }
                
                const secureMachineId = (params.machineId && params.machineId.toUpperCase() !== "EXPIRED") ? params.machineId : "";
                if (secureMachineId) {
                    sheet.getRange(i+1, COL_MACHINE_ID).setValue(secureMachineId); 
                }
                
                sheet.getRange(i+1, COL_STATUS).setValue('LICENSE ACTIVE');
                
                logSecurityAudit(ss, "LICENSE_ACTIVATION", "SUCCESS", "License Activated for ID " + params.userID, params);
                
                // --- ADDED: Activation Success Emails ---
                const activationData = {
                  userName: data[i][1],
                  userID: String(data[i][2]).toUpperCase(),
                  attemptMachine: secureMachineId,
                  regDate: data[i][6] || "N/A",
                  dataSize: (data[i][10] || 5000) + " Emps",
                  expiryDate: data[i][7]
                };
                if (isSyncActivation) {
                    sendEmailNotification(data[i][5], "Credentials Synchronized", "CREDENTIALS_SYNCED", { userName: data[i][1] });
                } else {
                    sendEmailNotification(data[i][5], "✅ License Activated: BharatPay Pro", "LICENSE_ACTIVATED", activationData);
                    sendEmailNotification("ilcbala.BharatPayRoll@gmail.com", "🚀 License Activated: " + data[i][5], "LICENSE_ACTIVATED", activationData);
                }

                if (lock.hasLock()) lock.releaseLock();
                return jsonResponse({ 
                    success: true, 
                    data: { 
                        status: 'LICENSE ACTIVE', 
                        expiryDate: data[i][7], 
                        dataSize: data[i][10] || 5000,
                        password: data[i][3]
                    } 
                });
            }
        }
        logSecurityAudit(ss, "LICENSE_ACTIVATION", "FAILURE", "Key/ID mismatch for " + params.userID, params);
        if (lock.hasLock()) lock.releaseLock();
        return jsonResponse({ success: false, message: "Invalid Credentials. Please verify your License Key and User ID." });
    } catch (e) {
        if (lock.hasLock()) lock.releaseLock();
        logSecurityAudit(ss, "LICENSE_ACTIVATION", "CRITICAL_ERROR", e.toString(), params);
        return jsonResponse({ success: false, message: e.toString() });
    }
}

function handleTrackLogin(ss, params) {
    if (IsDeveloperRequest(ss, params)) return jsonResponse({ success: true, note: "Dev Isolation Mode" });

    const now = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy HH:mm:ss");
    ['BPP_APP_License', 'Approved_Trial_Data'].forEach(function(name) {
        const sheet = ss.getSheetByName(name); if (!sheet) return;
        const isTrial = name.includes('Trial');
        
        const emailColIdx = isTrial ? 3 : 5;
        const machColIdx  = isTrial ? 5 : 8;
        
        const data = sheet.getDataRange().getValues();
        for (let i = 1; i < data.length; i++) {
            const rowEmail = String(data[i][emailColIdx]).toLowerCase().trim();
            const rowMach  = String(data[i][machColIdx]).toLowerCase().trim();
            
            if (rowEmail === params.email.toLowerCase().trim() && (rowMach === "" || rowMach === params.machineId.toLowerCase().trim())) {
                const currentCount = parseInt(data[i][COL_LOGIN_COUNT - 1]) || 0;
                const loginCell = sheet.getRange(i+1, COL_LOGIN_COUNT);
                loginCell.setNumberFormat('0');
                loginCell.setValue(currentCount + 1);
                
                const lastLoginCell = sheet.getRange(i+1, COL_LAST_LOGIN);
                lastLoginCell.setValue(now);
                
                sheet.getRange(i+1, COL_TOTAL_HOURS).setNumberFormat('[h]:mm');
                break;
            }
        }
    });
    return jsonResponse({ success: true });
}

function repairLicenseDatabase() { 
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  cleanupStaleActiveUsers(ss);
  updateExpiryStatuses(ss); 
  
  const todayStr = Utilities.formatDate(new Date(), "GMT+5:30", "yyyyMMdd");

  ['BPP_APP_License', 'Approved_Trial_Data'].forEach(function(name) {
    const sheet = ss.getSheetByName(name); if (!sheet) return;
    const data = sheet.getDataRange().getValues();
    const isTrial = name.includes('Trial');
    const mColIdx = isTrial ? 5 : 8; // Audited: Trial=Col F(5), License=Col I(8)
    const sColNum = 10; // Status=Col J
    
    sheet.getRange(2, COL_LOGIN_COUNT, sheet.getLastRow(), 1).setNumberFormat('0');
    sheet.getRange(2, COL_TOTAL_HOURS, sheet.getLastRow(), 1).setNumberFormat('[h]:mm');
    
    for (let i = 1; i < data.length; i++) {
        const rowData = data[i];
        const rawMachine = String(rowData[mColIdx]).trim();
        const expStr = String(rowData[7]);
        const p = expStr.split("-");
        const isCurrentlyExpired = (p.length === 3 && (p[2]+p[1]+p[0] < todayStr));

        // 1. If Machine ID column mistakenly contains a status string, move it
        const upperMach = rawMachine.toUpperCase();
        if (upperMach === "EXPIRED" || upperMach === "PENDING" || upperMach.includes("ACTIVE")) {
            sheet.getRange(i+1, mColIdx + 1).setValue(""); // Clear Machine ID Col
            const finalStatus = isCurrentlyExpired ? (isTrial ? "TRIAL_EXPIRED" : "LICENSE_EXPIRED") : (isTrial ? "TRIAL ACTIVE" : "LICENSE ACTIVE");
            sheet.getRange(i+1, sColNum).setValue(finalStatus);
            continue;
        }

        // 2. Refresh based on existing Machine ID and Date
        if (isCurrentlyExpired) {
            sheet.getRange(i+1, sColNum).setValue(isTrial ? "TRIAL_EXPIRED" : "LICENSE_EXPIRED");
        } else if (rawMachine === "" || rawMachine === "UNKNOWN-MACHINE-ID") {
            sheet.getRange(i+1, sColNum).setValue("PENDING");
        } else {
            sheet.getRange(i+1, sColNum).setValue(isTrial ? "TRIAL ACTIVE" : "LICENSE ACTIVE");
        }
    }
  });

  const activeSheet = ss.getSheetByName('Active_Users');
  if (activeSheet) {
    const activeData = activeSheet.getDataRange().getValues();
    const seenMachines = {};
    const rowsToDelete = [];
    for (let i = activeData.length - 1; i >= 1; i--) {
      const mid = String(activeData[i][1]);
      if (seenMachines[mid]) rowsToDelete.push(i + 1);
      else seenMachines[mid] = true;
    }
    rowsToDelete.sort((a,b) => b-a).forEach(row => activeSheet.deleteRow(row));
  }
  SpreadsheetApp.getUi().alert("🛠️ Full Database Repair Complete: Status Labels & Column alignment restored."); 
}

function generateHardenedPassword() {
  const upper = "ABCDEFGHJKLMNPQRSTUVWXYZ";
  const lower = "abcdefghijkmnopqrstuvwxyz";
  const nums = "23456789";
  const spec = "!@#$%^&*()_+";
  const all = upper + lower + nums + spec;
  
  let pass = "";
  pass += upper[Math.floor(Math.random() * upper.length)];
  pass += lower[Math.floor(Math.random() * lower.length)];
  pass += nums[Math.floor(Math.random() * nums.length)];
  pass += spec[Math.floor(Math.random() * spec.length)];
  
  for (let i = 0; i < 5; i++) {
    pass += all[Math.floor(Math.random() * all.length)];
  }
  
  return pass.split('').sort(() => 0.5 - Math.random()).join('');
}

/**
 * Cleans up 'Ghost' sessions in Active_Users sheet.
 */
function cleanupStaleActiveUsers(ss) {
    const sheet = ss.getSheetByName('Active_Users');
    if (!sheet) return;
    const data = sheet.getDataRange().getValues();
    const now = new Date().getTime();
    const threshold = 10 * 60 * 1000; // 10 minutes

    for (let i = 1; i < data.length; i++) {
        const status = String(data[i][3]).toUpperCase();
        if (status === 'LIVE') {
            const lastActiveStr = String(data[i][4]);
            const lastActiveDate = parseManualDate(lastActiveStr);
            if (lastActiveDate && (now - lastActiveDate.getTime()) > threshold) {
                sheet.getRange(i + 1, 4).setValue('DISCONNECTED')
                    .setBackground("#fee2e2")
                    .setFontColor("#991b1b")
                    .setFontWeight("bold");
            }
        }
    }
}

function parseManualDate(str) {
    try {
        const p = str.match(/(\d+)-(\d+)-(\d+) (\d+):(\d+):(\d+)/);
        if (!p) return null;
        const iso = p[3] + "-" + p[2] + "-" + p[1] + "T" + p[4] + ":" + p[5] + ":" + p[6] + "+05:30";
        return new Date(iso);
    } catch (e) { return null; }
}

/**
 * Helper: Compares two version strings (e.g. "02.02.21" vs "01.01.12")
 * Returns: 1 if v1 > v2, -1 if v1 < v2, 0 if equal.
 */
function compareVersions(v1, v2) {
    var n1 = v1.replace(/^[vV]/, '').split('.').map(Number);
    var n2 = v2.replace(/^[vV]/, '').split('.').map(Number);
    for (var i = 0; i < Math.max(n1.length, n2.length); i++) {
        var a = n1[i] || 0;
        var b = n2[i] || 0;
        if (a > b) return 1;
        if (a < b) return -1;
    }
    return 0;
}

                const dbMobile = String(data[i][mCol]).trim();
                if (dbMobile === mobile) {
                    foundIdentity = { 
                        userID: String(data[i][uCol]).toUpperCase(), 
                        userName: String(data[i][nCol]).toUpperCase() 
                    };
                } else if (mobile) {
                    mobileMismatch = true;
                }
                break;
            }
        }
    });

    if (mobileMismatch && !foundIdentity) {
        return jsonResponse({ success: false, message: "Security Mismatch: The provided Mobile Number does not match our registered records for this email." });
    }

    // Preserve OTP for the final activation step
    PropertiesService.getScriptProperties().setProperty('OTP_' + email, otp);
    PropertiesService.getScriptProperties().deleteProperty(otpKey);
    PropertiesService.getScriptProperties().deleteProperty(timeKey);

    return jsonResponse({ 
        success: true, 
        message: "Email and Identity Verified successfully.",
        data: foundIdentity 
    });
}


/**
 * V02.02.21: TOTAL SECURITY LOCKDOWN
 * Programmatically protects ALL sheets in the spreadsheet, ensuring only 
 * the developer (script owner) can manually edit data.
 */
function protectSensitiveSheets() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheets = ss.getSheets();
  const userEmail = Session.getEffectiveUser().getEmail();
  
  sheets.forEach(sheet => {
    const name = sheet.getName();
    // Special handling for Active_Users and Security_Audit might be lower-priority but we lock them too
    const protection = sheet.protect().setDescription('BPP Hardened Shield — ' + name);
    
    // V02.02.21 FIX: Explicitly disable warning-only mode to allow editor restriction
    protection.setWarningOnly(false);
    
    // Ensure only the script owner (developer) can edit
    protection.removeEditors(protection.getEditors());
    if (protection.canEdit()) {
      try {
        protection.addEditor(userEmail);
      } catch (e) {
        console.warn("Could not add editor to " + name + ": " + e.toString());
      }
    }
    
    // If there are other specific owners/admins you want to add, you'd add them here
    console.log("🛡️ Protected Sheet: " + name);
  });
  
  ss.toast("🛡️ All sheets are now programmatically locked. Only you (" + userEmail + ") can edit.", "BPP Total Security", 7);
}
// ── AUTOMATIC HOT-PATCH TRIGGER ──
// Updates Column F (Timestamp) whenever B (Win10 Link), C (Win7 Link), or E (Patch Context) changes.
function onEdit(e) {
  const sheet = e.source.getActiveSheet();
  if (sheet.getName() !== 'App_Config') return;
  
  const range = e.range;
  const col = range.getColumn();
  const row = range.getRow();
  
  // Only watch row 2, columns B(2), C(3), and E(5)
  if (row === 2 && (col === 2 || col === 3 || col === 5)) {
    const timestamp = Utilities.formatDate(new Date(), "GMT+5:30", "dd-MM-yyyy HH:mm:ss");
    sheet.getRange(2, 6).setValue(timestamp); // Update Column F
    console.log("🛠️ App_Config Timestamp Updated: " + timestamp);
  }
}
